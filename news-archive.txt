<?php
/*
* Template Name: Shablon news archie
* Template post type: news
  */
?>

<?php
if (!defined('ABSPATH')) {
    exit; // Exit if accessed directly.
}
get_header('html'); ?>
<?php
$current_page = !empty($_GET['nnews']) ? $_GET['nnews'] : 1;
$query = new WP_Query(array(
    'post_type' => 'news',
    'posts_per_page' => 10,
    'paged' => $current_page,
));
?>

<div class="wrapper">
    
    <div class="s-news-arh-wrap">
        <div class="container">

        <h1 class="news-page__title page-title">Новости</h1>
        <?php
        if (have_posts()) {
            ?>
            <div class="s-news-arh-blocks">
                <?php while ($query->have_posts()) : $query->the_post();
                    ?>
                    <div class="s-news-arh-block">
                        <div class="s-news-arh-block__img">
                            <a href="<?php the_permalink(); ?>">
                                <img src="<?php echo eko_get_post_image_src(get_the_ID()); ?>" alt="">
                            </a>
                        </div>
                        <div class="s-news-arh-block__content">
                            <div class="s-news-arh-block__date news-page-item__date">
	                            <?php echo eko_get_post_date(); ?>
                            </div>
                            <div class="s-news-arh-block__main">
                                <h2 class="news-page-item__title"><a href="<?php the_permalink(); ?>"><?php the_title(); ?></a></h2>
                            </div>
                            <div class="s-news-arh-block__text">
                                <?php echo get_field('news_anons', get_the_ID()); ?>
                            </div>
                        </div>
                    </div>

                <?php endwhile;
                ?></div>
            <div class="pagination">
                <?php

                echo paginate_links(array(
                    'prev_text' => '&larr; ',
                    'next_text' => ' &rarr;',
                    'base' => site_url() . '/news/' . '%_%',
                    'format' => '?nnews=%#%',
                    'total' => $query->max_num_pages,
                    'current' => $current_page,
                ));
                ?></div>
            <?php
        }
        wp_reset_postdata();
        ?>
    </div>
    </div>

</div>
<?php get_footer('html'); ?>
